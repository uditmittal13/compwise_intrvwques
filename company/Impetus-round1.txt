I have a collection of numbers in an array and want to find the matching pair that is equal to the given sum.
Example1: Array={1, 2, 4, 5}  sum = 9
Example2: Array={1, 2, 3, 5}  sum = 9


boolean isPairExists(int[] arr,int sum){
	int calSum=0;
	for(int prevIndex=0;prevIndex<arr.length;prevIndex++){
		for(int index=prevIndex+1;index<arr.length;index++){
			calSum= arr[prevIndex]+arr[index];
		if(calSum==sum){
			return true;
		}
	   }
	}
	return false;
}
